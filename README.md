###git操作
--1,设定名称和邮箱
git config --global user.name "xxxx" 
git config --global user.email xxxxx@163.com
--2,创建本地仓库
git init
--3,上传所以文件到本地仓库
git add.
--4,提交
git commit -m "initial commit"
--5,关联本地仓库并上传代码
git remote add origin https://github.com/XXX/XXXXX
--6,push
git push origin master
